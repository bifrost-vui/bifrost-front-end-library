## ICardContentLists

<h3 id="icardcontentlists-api">API</h3>

<ArgsTable of={CardContentListsAPI} />

<h3 id="icardcontentlists-format">Expected format</h3>

```jsx
{
    "channels": {
        full: [
            {
                "tabContent": {
                    "ariaLabelledby": "tabIncluded",
                    "channelsList": [
                        {
                            "imageSrc": "/images/_docs/channels/CFCMH.webp",
                            "name": "TVA",
                        },
                        {...}
                    ],
                    "class": "show active",
                    "description": "<p><strong>23 channels included</strong></p>",
                    "id": "tabContentIncluded",
                },
                "tabData": {
                    "ariaControls": "tabContentIncluded",
                    "ariaSelected": "true",
                    "class": "active",
                    "href": "tabContentIncluded",
                    "id": "tabIncluded",
                    "label": "Included",
                },
            },
            {...}
        ],
        limited: [
            {
                "imageSrc": "/images/_docs/channels/CFCMH.webp",
                "name": "TVA",
            },
            {...}
        ],
    },
    "content": [
        {
            "icon": "unlimited",
            "label": "Free and Unlimited Go after 10Go",
        },
        {...},
    ],
    "title": "TV",
    "description": "40 channels",
}
```
