                    ],
                    "class": "show active",
                    "description": "<p><strong>23 channels included</strong></p>",

                "tabData": {

            {...}
        ],

            {
                "imageSrc": "/images/_docs/channels/CFCMH.webp",
                "name": "TVA",
            },
            {...}

        {
            "icon": "unlimited",
