    bfSlick.assignCarousel('.js-bf-carousel-demo', {

        centerPadding: "48px",
        dots: true,
        infinite: false,
        mobileFirst: true,
        nextArrow: bfSlick.getNextSlickBtn(),

        slidesToScroll: 1,
        responsive: [
            {
                breakpoint: bfTheme.breakpoints.sm - 1,
                settings: {
                    arrows: true,
                    centerMode: false,
                    slidesToShow: 2,
                    slidesToScroll: 2,
