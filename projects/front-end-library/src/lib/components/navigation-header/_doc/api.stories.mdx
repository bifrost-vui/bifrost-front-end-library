## IMenuItem

This API extends <a href="#idefaultmenuitem-api" target="_self">IDefaultMenuItem API</a>.
All props from IDefaultMenuItem API are available, plus the ones below.

<h3 id="imenuitem-api">API</h3>

<ArgsTable of={MenuItemAPI} />

<h3 id="imenuitem-format">Expected format</h3>

```jsx
{
    "color": "#FF0000",
    "class": "class-name",
    "href": "/moving",
    "isActive": false,
    "isActiveLinkClickable": false,
    "label": "Moving",
    "rel": "dofollow",
    "subItems": [
        {
            "class": "other-class-name",
            "color": "#00FF00",
            "href": "/lorem-ipsum",
            "isActive": false,
            "isActiveLinkClickable": false,
            "label": "Lorem Ipsum",
            "rel": "dofollow",
            "unclickableActiveMenuItemScreenReaderMessage": "Current page: ",
        },
        {...},
    ],
    "unclickableActiveMenuItemScreenReaderMessage": "Current page: ",
}
```


## ITopBar

<h3 id="itopbar-api">API</h3>

<ArgsTable of={TopBarAPI} />

<h3 id="itopbar-format">Expected format</h3>

```jsx
{
    "left": {
        "description": "Website Selection",
        "items": [
            {
                "href": "/",
                "isActive": true,
                "isActiveLinkClickable": false,
                "label": "Residential",
                "rel": "dofollow",
                "unclickableActiveMenuItemScreenReaderMessage": "Current website:",
            },
            {
                "href": "/",
                "label": "Business",
                "rel": "dofollow",
            },
            {...},
        ],
    },
    "right": {
        "description": "Other links",
        "items": [
            {
                "href": "/",
                "label": "Other link #1",
                "rel": "dofollow",
            },
            {
                "href": "/",
                "label": "Other link #2",
                "rel": "dofollow",
            },
            {...},
        ],
    },
}
```
